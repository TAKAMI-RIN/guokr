字符串参数解析QueryString

描述：
* 这是一个处理搜索字符串的库，比如用于获取url里面search部分的参数，或者获取hash部分的参数。只要是以key=value&key2=value2为格式的都可以搞定
* 参考: http://unixpapa.com/js/querystring.html
* @author mzhou
* @version 0.1
引用js和css：
JS:
//@import "lib/QueryString.js";
API接口
    $('#editDemo').click( function() {
        // ask.js
        /**
         * json转字符串,值为null或undefined或''的时候会不会被转换
         * @param {object} json json对象
         * @return {string} 字符串
         */
        var query   = QueryString.toQueryString(),
        /**
         * Query.getSearch
         * 获取url里面search部分的Query对象
         * @return {object} Query对象
         */
            search  = QueryString.getSearch(),
        /**
         * 获取url里面hash部分的Query对象
         * @return {object} Query对象
         */
            hash    = QueryString.getHash();
        /**
         * 获取某个key值，值为''时返回undefined
         *  ''        ==> undefined
         *  undefined ==> undefined
         *  null      ==> null
         *  0         ==> 0
         * 如果没有设置值，则统一返回undefined
         */
        if( search.get('_edit') ) {
            var sel = $(this).data('id');
            sel = '#demo' + sel;
            var $demo = $(sel);
            var hashJson = '';
            $.each( hash.dict, function(i,n){
                hashJson += i + ' : ' + n;
            });
            $demo.prepend( '通过data-id：' + sel + ' 选中，QueryString.getHash得到：' + hashJson + '<br/>');
        }
    }).click();     // demo需要，先跳转再实现修改
