随机函数库random

描述：
* 随机函数库
* 使用方法：
* random([1,3,4]) // 随机获取数组中的某一项
* random([1,3,4], 3) // 随机选取数组中前3个项中某一项
* random([1,3,4], function(item) { return item;}) // 以数组的值为权重，随机选取数组中的某一项
* random( 0, 10 ) // 随机0到10中的一个整数,包括0，但不包括0，即[0,10)
* random( 0, 10, true ) // 随机0到10中的非整数，包括0，但不包括0，即[0,10)
* @author mzhou
引用js和css：
JS:
//@import "lib/random.js";
计算指定区间之间的随机数
/**
 * 计算指定区间之间的随机数
 *      区间为[from,to)
 * @return {number} 
 */
var result = random(
    // @param {number} from 开始值
    1,
    // @param {number} to 结束值
    10,
    // @param {boolean} notInteger 是否返回非整数，默认为false
    true
);

随机选取数组中前几个项中某一项
/**
 * 随机选取数组中前几个项中某一项
 * @param {number} end 结束值
 * @return {object}
 */
var result = random(
    [1,3,4,6,7],
    // @param {number} end 结束值，可缺省
    4
);

以数组的值为权重，随机选取数组中的某一项
/**
 * 以数组的值为权重，随机选取数组中的某一项
 * @param {function} getWeight 获取weight的方式，默认为obj的weight属性。 权重值小于等于0则不会被选取到。
 * @return {object} 
 */
var result = random(
    [1,2,3,5,20,-5,-8,-7],
    // @param {function} getWeight 获取weight的方式，默认为obj的weight属性。 权重值小于等于0则不会被选取到。
    function( obj ){
        return obj;
    }
);
