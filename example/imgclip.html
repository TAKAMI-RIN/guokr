    var $crop = $('.crop');
    $crop.find('.preview_big_pic').bind( 'load', function() {
        imgclip(
            // headSelector, 图片容器对象
            $(this), {
            // previewSelectors, 预览对象
            '#preview160': [160, 160],
            '#preview48': [48, 48]
            }, {
            // option, 可选
            // 设置裁剪框的大小，默认为待裁剪图片的高度和宽度中的较小的一个值
            'width': 100,
            'height': 100,
            // 设置裁剪框离top 和left 距离，默认为0
            'left': 20,
            // 裁剪最小宽度和高度，默认为48
            'minWidth': 20,
            'minHeight': 20
            }
        );
    }).load();
