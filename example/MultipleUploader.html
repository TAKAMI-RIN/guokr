多文件上传MultipleUploader

描述：
* 多文件异步上传模块，用iframe实现
* @author: mzhou
* @version: 1.0
引用js和css：
JS:
//@import "lib/MultipleUploader.js";
API接口
var uploader1 = new MultipleUploader('#uploader1', '/image.json', {params: {ukey: 'adasda'}});
uploader1
    .on('success', function(id, data) {
        $(uploader1.getFileView(id))
            .append(G.format('<br/><mark>hashkey为{hashkey}，宽度为{width}，高度为{height}</mark>', data.result));
    });
/**
依据 CSS 钩子：
success状态为绿色
