API接口
slide(
    // 大看版的容器对象，此对象第一个元素为图片集合，第二个为触发器集合
    $('#slide'), 
    // 指定默认显示的图片，默认为0开始
    1,
    // 指定轮播间隔，默认为5000ms
    1000
);
